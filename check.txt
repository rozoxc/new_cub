bool	ends(char *s, char *end)
{
	int	i;

	// if (ft_strlen(end) > ft_strlen(s)) // check if it need to be >=
	//     return (false);
	i = 0;
	while (s[i])
	{
		if (s[i] == '.')
			break ;
		i++;
	}
	if (i == ft_strlen(s))
		return (false);
	else if (ft_strcmp(s + i, end) == 0)
		return (true);
	return (false);
}
